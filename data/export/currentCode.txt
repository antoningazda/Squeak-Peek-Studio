        function applyTheme(app, themeStruct)
            app.Theme = themeStruct;
            app.ThemeDropDown.Value = app.AppSettings.Other.ThemeName;            % update dropdown
            fig = app.SqueakPeekStudioUIFigure;
            fig.Color = themeStruct.Background;

            for i = 1:numel(app.ButtonsMenu)
                app.ButtonsMenu(i).BackgroundColor = themeStruct.Background;
                app.ButtonsMenu(i).FontColor = themeStruct.Text;
            end

            for i = 1:numel(app.ButtonsOther)
                btn = app.ButtonsOther(i);
                if isstruct(btn.UserData) && isfield(btn.UserData, "Style") && strcmp(btn.UserData.Style, "Primary")
                    btn.BackgroundColor = themeStruct.Primary;
                    btn.FontColor = themeStruct.Text;
                elseif isstruct(btn.UserData) && isfield(btn.UserData, "Style") && strcmp(btn.UserData.Style, "Accent")
                    btn.BackgroundColor = themeStruct.Accent;
                    btn.FontColor = themeStruct.Text;
                else
                    btn.BackgroundColor = themeStruct.Background;
                    btn.FontColor = themeStruct.Text;
                end

            end


            for i = 1:numel(app.Grids)
                app.Grids(i).BackgroundColor = themeStruct.Background;
            end
            for i = 1:numel(app.Labels)
                app.Labels(i).FontColor = themeStruct.Text;
            end


            app.ButtonsMenuGrid.BackgroundColor = themeStruct.Primary;
        end

        function loadSettingsFromFile(app, filepath)
            if nargin < 2
                [file, path] = uigetfile('*.json', 'Load Settings');
                if isequal(file, 0)
                    return; % User canceled
                end
                filepath = fullfile(path, file);
            end

            fid = fopen(filepath, 'r');
            if fid == -1
                uialert(app.SqueakPeekStudioUIFigure, 'Could not open file for reading.', 'File Error');
                return;
            end

            raw = fread(fid, '*char')';
            fclose(fid);

            try
                settings = jsondecode(raw);
                app.AppSettings = settings;
                ApplyAllAppSettings(app);  % Re-apply after loading
            catch ME
                uialert(app.SqueakPeekStudioUIFigure, sprintf('Error reading settings:\n%s', ME.message), 'JSON Error');
            end
        end

        function loadDefaultSettingsIfExists(app)
            preloadName = app.AppSettings.Other.DefaultSettingsFile;
            appFolder = fileparts(mfilename('fullpath'));
            settingsPath = fullfile(appFolder, 'settings', preloadName);

            if exist(settingsPath, 'file') == 2
                try
                    loadSettingsFromFile(app, settingsPath);
                    disp("Loaded default settings from: " + settingsPath);
                catch err
                    warning('%s: %s', err.identifier, err.message);
                end
            else
                warning("Default settings file not found at: %s", settingsPath);
            end
        end


        function [files, numFiles] = getFilesFromFolder(~,folderPath, fileExtension)
            % getFilesFromFolder - Lists files with the given extension in a folder.
            %
            % Inputs:
            %   folderPath     - string or char, path to the folder
            %   fileExtension  - string or char, e.g. '*.wav', '*.txt', '*.mov'
            %
            % Outputs:
            %   files     - struct array from dir()
            %   numFiles  - number of files found

            if nargin < 2
                error('You must provide both folderPath and fileExtension, e.g., ''*.wav''');
            end

            files = dir(fullfile(folderPath, fileExtension));
            numFiles = numel(files);

            fprintf("Found %d %s files in:\n%s\n", numFiles, fileExtension, folderPath);
            for i = 1:numFiles
                fullPath = fullfile(files(i).folder, files(i).name);
                fprintf("%s\n", fullPath);
            end
        end



        function count = checkFileCount(app, type)
            try
                switch type
                    case "USVs"
                        [files, count] = getFilesFromFolder(app, app.SelectUSVFieldBatch.Value, '*.wav');
                        app.BatchFilesInfo(1).USVFiles = files;
                        app.BatchFilesInfo(1).numUSVs = count;
                        app.USVBatchFound.Text = sprintf("(%d)", count);
                    case "Labels"
                        [files, count] = getFilesFromFolder(app, app.SelectLabelFieldBatch.Value, '*.txt');
                        app.BatchFilesInfo(1).LabelFiles = files;
                        app.BatchFilesInfo(1).numLabels = count;
                        app.LabelBatchFound.Text = sprintf("(%d)", count);
                    case "ReferenceLabels"
                        [files, count] = getFilesFromFolder(app, app.SelectReferenceLabelFieldBatch.Value, '*.txt');
                        app.BatchFilesInfo(1).ReferenceLabelFiles = files;
                        app.BatchFilesInfo(1).numReferenceLabels = count;
                        app.ReferenceLabelBatchFound.Text = sprintf("(%d)", count);
                    case "Videos"
                        [files, count] = getFilesFromFolder(app, app.SelectVideoFieldBatch.Value, '*.mov');
                        app.BatchFilesInfo(1).VideoFiles = files;
                        app.BatchFilesInfo(1).numVideos = count;
                        app.VideoBatchFound.Text = sprintf("(%d)", count);
                    otherwise
                        warning("Wrong input for function checkFileCount");
                        count = 0;
                end
            catch ME
                warning("Error in checkFileCount for %s: %s", type, ME.message);
                count = 0;
            end
        end
    end

                        [usv, fs] = audioread(usvPath);

                        refLabel = importLabels(refLabelPath, fs);

        % Button pushed function: PlotButton
        function PlotButtonPushed(app, event)
            % --- Show loading dialog
            if app.AppSettings.Visualization.ShowLoadingDialog
                d = uiprogressdlg(app.SqueakPeekStudioUIFigure, ...
                    'Title', 'Please Wait', ...
                    'Message', 'Rendering waveform and spectrogram...', ...
                    'Indeterminate', 'on');
                drawnow;
            end

            try
                % --- Extract inputs
                USVVariable = app.CurrentInputFile.USV;
                labelStruct = app.CurrentInputFile.Label;
                referenceLabelStruct = app.CurrentInputFile.ReferenceLabel;
                fs = app.CurrentInputFile.fs;

                % --- Segment timing
                segmentStartSeconds = app.SegmentStartsEditField.Value;
                segmentLengthSeconds = app.SegmentLengthsEditField.Value;
                segmentEndSeconds = segmentStartSeconds+segmentLengthSeconds;

                % --- Index range
                segmentStartIndex = round(segmentStartSeconds * fs) + 1;
                segmentEndIndex   = round((segmentStartSeconds + segmentLengthSeconds) * fs);

                % --- Extract segment and generate time vector
                USVSegment = USVVariable(segmentStartIndex:segmentEndIndex);
                segmentTime = linspace(segmentStartSeconds, ...
                    segmentStartSeconds + segmentLengthSeconds, ...
                    length(USVSegment));

                % === Plot waveform ===
                plot(app.WaveformUIAxes, segmentTime, USVSegment);
                axis(app.WaveformUIAxes, 'tight');
                grid(app.WaveformUIAxes, 'on');
                xlabel(app.WaveformUIAxes, 'Time (s)');
                ylabel(app.WaveformUIAxes, 'Amplitude');
                xlim(app.WaveformUIAxes, [segmentStartSeconds, segmentStartSeconds + segmentLengthSeconds]);

                % === Compute and plot spectrogram ===
                windowSize = app.AppSettings.Visualization.SpectrogramWindow;
                overlap = app.AppSettings.Visualization.SpectrogramOverlap;

                [~, f, t, p] = spectrogram(USVSegment, windowSize, overlap, [], fs, 'yaxis');
                f_kHz = f / 1000;
                p_dB = 10 * log10(abs(p) + eps);  % avoid log(0)

                imagesc(app.SpectrogramUIAxes, t + segmentStartSeconds, f_kHz, p_dB);
                axis(app.SpectrogramUIAxes, 'xy');
                xlabel(app.SpectrogramUIAxes, 'Time (s)');
                ylabel(app.SpectrogramUIAxes, 'Frequency (kHz)');
                colormap(app.SpectrogramUIAxes, app.AppSettings.Visualization.SpectrogramColormap);
                ylim(app.SpectrogramUIAxes, [app.AppSettings.Visualization.SpectrogramMinFrequency, ...
                    app.AppSettings.Visualization.SpectrogramMaxFrequency]);
                xlim(app.SpectrogramUIAxes, [segmentStartSeconds, segmentStartSeconds + segmentLengthSeconds]);

                % === Set identical x-ticks ===
                xtickVals = linspace(segmentStartSeconds, segmentStartSeconds + segmentLengthSeconds, 11);
                set(app.WaveformUIAxes, 'XTick', xtickVals);
                set(app.SpectrogramUIAxes, 'XTick', xtickVals);


                %%%%% LABELS %%%%%
                minFreq = app.AppSettings.Visualization.SpectrogramMinFrequency;
                maxFreq = app.AppSettings.Visualization.SpectrogramMaxFrequency;
                showLabels = app.AppSettings.Visualization.ShowLabels;
                showRefLabels = app.AppSettings.Visualization.ShowReferenceLabels;

                % Clear previous label overlays
                hold(app.SpectrogramUIAxes, 'on');
                delete(findall(app.SpectrogramUIAxes, 'Tag', 'LabelOverlay'));

                % Plot Detected Labels
                labelsColor = app.AppSettings.Visualization.LabelColor;
                referenceLabelsColor = app.AppSettings.Visualization.ReferenceLabelColor;

                if showLabels
                    for i = 1:length(app.CurrentInputFile.Label)
                        L = app.CurrentInputFile.Label(i);
                        if L.StartTime >= segmentStartSeconds && L.EndTime <= segmentEndSeconds
                            x1 = L.StartTime; x2 = L.EndTime;
                            % Vertical lines
                            line(app.SpectrogramUIAxes, [x1 x1], [minFreq maxFreq], ...
                                'Color', labelsColor, 'LineWidth', 1.2, 'LineStyle', '--', 'Tag', 'LabelOverlay');
                            line(app.SpectrogramUIAxes, [x2 x2], [minFreq maxFreq], ...
                                'Color', labelsColor, 'LineWidth', 1.2, 'LineStyle', '-.', 'Tag', 'LabelOverlay');
                            % Text
                            if ~strcmp(L.Label, "None")
                                text(app.SpectrogramUIAxes, (x1 + x2) / 2, maxFreq - 3, L.Label, ...
                                    'Color', labelsColor, 'BackgroundColor', 'black', 'FontSize', 12, ...
                                    'HorizontalAlignment', 'center', 'Tag', 'LabelOverlay');
                            end
                        end
                    end
                end

                % Plot Reference Labels
                if showRefLabels
                    for i = 1:length(app.CurrentInputFile.ReferenceLabel)
                        R = app.CurrentInputFile.ReferenceLabel(i);
                        if R.StartTime >= segmentStartSeconds && R.EndTime <= segmentEndSeconds
                            x1 = R.StartTime; x2 = R.EndTime;
                            % Vertical lines
                            line(app.SpectrogramUIAxes, [x1 x1], [minFreq maxFreq], ...
                                'Color', referenceLabelsColor, 'LineWidth', 1.2, 'LineStyle', '--', 'Tag', 'LabelOverlay');
                            line(app.SpectrogramUIAxes, [x2 x2], [minFreq maxFreq], ...
                                'Color', referenceLabelsColor, 'LineWidth', 1.2, 'LineStyle', '-.', 'Tag', 'LabelOverlay');
                            % Text
                            if ~strcmp(R.Label, "None")
                                text(app.SpectrogramUIAxes, (x1 + x2) / 2, maxFreq - 8, R.Label, ...
                                    'Color', referenceLabelsColor, 'BackgroundColor', 'black', 'FontSize', 12, ...
                                    'HorizontalAlignment', 'center', 'Tag', 'LabelOverlay');
                            end
                        end
                    end
                end

                hold(app.SpectrogramUIAxes, 'off');

            catch ME
                warning(ME.identifier, 'Visualization error: %s', ME.message);
            end



            % --- Close loading dialog
            if exist('d', 'var') && isvalid(d)
                pause(0.2);
                close(d);
            end




        end

        % Button pushed function: RunDetectorsButton
        function RunDetectorsButtonPushed(app, event)
            % Show progress dialog
            d = uiprogressdlg(app.SqueakPeekStudioUIFigure, ...
                'Title', 'Running Detectors', ...
                'Message', 'Preparing...', ...
                'Indeterminate', 'on');

            drawnow;

            selectedDetectors = app.UsedDetectorsListBox.Value;
            selectedPostprocessing = app.UsedPostProcessingListBox.Value;

            if app.AppSettings.DataInput.BatchMode == false % single mode

                fs = app.CurrentInputFile.fs;
                usv = app.CurrentInputFile.USV;
                filename = app.CurrentInputFile.USVPath;

                exportDir = app.AppSettings.Detection.ExportPath;
                [~, baseName, ~] = fileparts(filename);  % Strip extension

                for k = 1:length(selectedDetectors)
                    det = selectedDetectors{k};
                    labelStruct = [];

                    switch det
                        case 'PSD'
                            d.Message = sprintf('Running PSD Detector on:\n%s', filename);
                            drawnow;
                            labelStruct = PSDDetector(usv, fs);

                        case 'BSCD'
                            d.Message = sprintf('Running BSCD Detector on:\n%s', filename);
                            drawnow;
                            labelStruct = BSCDDetector(usv, fs);  % Uncomment when ready

                        case 'RBD'
                            d.Message = sprintf('Running RBD Detector on:\n%s', filename);
                            drawnow;
                            labelStruct = RBD(usv, fs);  % Uncomment when ready
                    end

                    % Apply selected post-processing steps
                    for j = 1:length(selectedPostprocessing)
                        post = selectedPostprocessing{j};
                        switch post
                            case 'Merge Close Labels'
                                labelStruct = mergeCloseLabels(labelStruct, 0.001); % 1 ms
                                % Add more post-processing steps as needed
                            case 'Remove Short Labels'
                                labelStruct = removeShortLabels(labelStruct, 0.005); % 5 ms
                        end
                    end

                    % Export to file
                    exportFile = fullfile(exportDir, [baseName, '_', det, '_detected.txt']);
                    exportLabelsDetector(labelStruct, exportFile);
                end
            end

            % Close dialog
            if isvalid(d)
                close(d);
            end
        end

        % Button pushed function: LoadLabelEditButton
        function LoadLabelEditButtonPushed(app, event)
            % Update maximum label count
            app.LabelEditAllLabels.Value = length(app.CurrentInputFile_AR.Label);

            % Get current label index
            i = app.LabelEditCurrentLabel.Value;

            % Validate index
            if i < 1 || i > length(app.CurrentInputFile_AR.Label)
                warning("Invalid label index selected.");
                return;
            end

            % Get label struct and compute midpoint
            L = app.CurrentInputFile_AR.Label(i);
            midpoint = (L.StartTime + L.EndTime) / 2;
            segmentStartSeconds = midpoint - 0.25;
            segmentEndSeconds = midpoint + 0.25;

            % Clamp if outside signal
            fs = app.CurrentInputFile_AR.fs;
            USV = app.CurrentInputFile_AR.USV;
            totalDuration = length(USV) / fs;

            segmentStartSeconds = max(segmentStartSeconds, 0);
            segmentEndSeconds = min(segmentEndSeconds, totalDuration);
            segmentLengthSeconds = segmentEndSeconds - segmentStartSeconds;

            % Compute indices and extract audio segment
            segmentStartIndex = round(segmentStartSeconds * fs) + 1;
            segmentEndIndex = round(segmentEndSeconds * fs);
            USVSegment = USV(segmentStartIndex:segmentEndIndex);

            % Spectrogram calculation
            windowSize = app.AppSettings.Visualization.SpectrogramWindow;
            overlap = app.AppSettings.Visualization.SpectrogramOverlap;

            [~, f, t, p] = spectrogram(USVSegment, windowSize, overlap, [], fs, 'yaxis');
            f_kHz = f / 1000;
            p_dB = 10 * log10(abs(p) + eps);  % avoid log(0)

            % Plot
            imagesc(app.LabelEditSpectrogramUIAxes, t + segmentStartSeconds, f_kHz, p_dB);
            axis(app.LabelEditSpectrogramUIAxes, 'xy');
            xlabel(app.LabelEditSpectrogramUIAxes, 'Time (s)');
            ylabel(app.LabelEditSpectrogramUIAxes, 'Frequency (kHz)');
            colormap(app.LabelEditSpectrogramUIAxes, app.AppSettings.Visualization.SpectrogramColormap);
            ylim(app.LabelEditSpectrogramUIAxes, [app.AppSettings.Visualization.SpectrogramMinFrequency, ...
                app.AppSettings.Visualization.SpectrogramMaxFrequency]);
            xlim(app.LabelEditSpectrogramUIAxes, [segmentStartSeconds, segmentEndSeconds]);

            % Add label overlay
            hold(app.LabelEditSpectrogramUIAxes, 'on');
            line(app.LabelEditSpectrogramUIAxes, [L.StartTime, L.StartTime], ylim(app.LabelEditSpectrogramUIAxes), ...
                'Color', 'w', 'LineWidth', 1.2, 'LineStyle', '--');
            line(app.LabelEditSpectrogramUIAxes, [L.EndTime, L.EndTime], ylim(app.LabelEditSpectrogramUIAxes), ...
                'Color', 'w', 'LineWidth', 1.2, 'LineStyle', '-.');
            hold(app.LabelEditSpectrogramUIAxes, 'off');

            app.LabelClassificationEditField.Value = L.Label;
        end

        % Button pushed function: RejectButton
        function RejectButtonPushed(app, event)
            i = app.LabelEditCurrentLabel.Value;

            % Delete the label
            app.CurrentInputFile_AR.Label(i) = [];

            % Update total label count
            app.LabelEditAllLabels.Value = length(app.CurrentInputFile_AR.Label);

            % If we're at the end, move back, else move forward
            if i > length(app.CurrentInputFile_AR.Label)
                app.LabelEditCurrentLabel.Value = max(i - 1, 1);
            end

        % Button pushed function: ExportLabelsButton
        function ExportLabelsButtonPushed(app, event)
            [fileName, folderPath] = uiputfile('*.txt', 'Save Labels As', 'exported_labels.txt');
            if isequal(fileName, 0) || isequal(folderPath, 0)
                return; % User cancelled
            end

            fullPath = fullfile(folderPath, fileName);
            exportLabels(app.CurrentInputFile_AR.Label, fullPath);
        end

        % Button pushed function: SelectDetectorsExportFolder
        function SelectDetectorsExportFolderButtonPushed(app, event)
            folderPath = uigetdir();
            if isequal(folderPath, 0)
                return; % User canceled
            end
            app.SelectDetectorsExportFolderEditField.Value = folderPath;
        end

        function SelectDetectorsExportFolderEditFieldValueChanged(app, event)
            value = app.SelectDetectorsExportFolderEditField.Value;
            app.AppSettings.Detection.ExportPath = value;
        end

    % App creation and deletion
    methods (Access = public)

        % Construct app
        function app = SqueakPeekStudio

            runningApp = getRunningApp(app);

            % Check for running singleton app
            if isempty(runningApp)

                % Create UIFigure and components
                createComponents(app)

                % Register the app with App Designer
                registerApp(app, app.SqueakPeekStudioUIFigure)

                % Execute the startup function
                runStartupFcn(app, @startupFcn)
            else

                % Focus the running singleton app
                figure(runningApp.SqueakPeekStudioUIFigure)

                app = runningApp;
            end

            if nargout == 0
                clear app
            end
        end

        % Code that executes before app deletion
        function delete(app)

            % Delete UIFigure when app is deleted
            delete(app.SqueakPeekStudioUIFigure)
        end
    end
